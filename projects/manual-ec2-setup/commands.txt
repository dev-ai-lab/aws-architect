
# DBName=database name for wordpress
# DBUser=mariadb user for wordpress
# DBPassword=password for the mariadb user for wordpress
# DBRootPassword = root password for mariadb

# STEP 1 - Configure Authentication Variables which are used below
DBName='a4lwordpress'
DBUser='a4lwordpress'
DBPassword='4n1m4l$4L1f3'
DBRootPassword='4n1m4l$4L1f3'

# STEP 2 - Install system software - including Web and DB
sudo dnf install wget php-mysqlnd httpd php-fpm php-mysqli mariadb105-server php-json php php-devel -y


# STEP 3 - Web and DB Servers Online - and set to startup. Using these commands they are set to start now and on reboot.

sudo systemctl enable httpd
sudo systemctl enable mariadb
sudo systemctl start httpd
sudo systemctl start mariadb


# STEP 4 - Set Mariadb Root Password using the variable set above
sudo mysqladmin -u root password $DBRootPassword


# STEP 5 - Install Wordpress
## download and unzip latest wordpress to /var/www/html (web root)
sudo wget http://wordpress.org/latest.tar.gz -P /var/www/html
## move to the folder and unzip
cd /var/www/html
sudo tar -zxvf latest.tar.gz
## copy all files from wordpress folder to web root, then clean up
sudo cp -rvf wordpress/* .
sudo rm -R wordpress
sudo rm latest.tar.gz


# STEP 6 - Configure Wordpress
## create wp-config.php from sample file and set DB variables. Use sed to replace the placeholders in the sample file with the variables set above.
sudo cp ./wp-config-sample.php ./wp-config.php
sudo sed -i "s/'database_name_here'/'$DBName'/g" wp-config.php
sudo sed -i "s/'username_here'/'$DBUser'/g" wp-config.php
sudo sed -i "s/'password_here'/'$DBPassword'/g" wp-config.php
## set ownership of all files to apache user which is the web server user
sudo chown apache:apache * -R


# STEP 7 Create Wordpress DB. We need to create the database and user for wordpress to use. We will create a temp file with the commands and then run it using the mysql command line tool.
## Steps are: create database, create user, grant privileges and flush privileges
echo "CREATE DATABASE $DBName;" >> /tmp/db.setup
echo "CREATE USER '$DBUser'@'localhost' IDENTIFIED BY '$DBPassword';" >> /tmp/db.setup
echo "GRANT ALL ON $DBName.* TO '$DBUser'@'localhost';" >> /tmp/db.setup
echo "FLUSH PRIVILEGES;" >> /tmp/db.setup
cat /tmp/db.setup
## run the commands using the mysql command line tool
mysql -u root --password=$DBRootPassword < /tmp/db.setup
## clean up temp file
sudo rm /tmp/db.setup
